package com.calculator.service;

import com.calculator.exception.DivisionByZeroException;
import com.calculator.exception.NegativeSqrtException;
import org.springframework.stereotype.Service;

/**
 * Core calculation logic. All operations are performed on {@code double} values.
 * Throws domain-specific exceptions instead of returning NaN or Infinity.
 */
@Service
public class CalculatorService {

    /**
     * Adds two numbers.
     *
     * @param a first operand
     * @param b second operand
     * @return a + b
     */
    public double add(double a, double b) {
        return a + b;
    }

    /**
     * Subtracts {@code b} from {@code a}.
     *
     * @param a minuend
     * @param b subtrahend
     * @return a - b
     */
    public double subtract(double a, double b) {
        return a - b;
    }

    /**
     * Multiplies two numbers.
     *
     * @param a first factor
     * @param b second factor
     * @return a * b
     */
    public double multiply(double a, double b) {
        return a * b;
    }

    /**
     * Divides {@code a} by {@code b}.
     *
     * @param a dividend
     * @param b divisor — must not be zero
     * @return a / b
     * @throws DivisionByZeroException if {@code b} is zero
     */
    public double divide(double a, double b) {
        if (b == 0) {
            throw new DivisionByZeroException();
        }
        return a / b;
    }

    /**
     * Returns the square root of {@code a}.
     *
     * @param a the radicand — must be non-negative
     * @return √a
     * @throws NegativeSqrtException if {@code a} is negative
     */
    public double sqrt(double a) {
        if (a < 0) {
            throw new NegativeSqrtException(a);
        }
        return Math.sqrt(a);
    }

    /**
     * Raises {@code base} to the power of {@code exponent}.
     *
     * @param base     the base
     * @param exponent the exponent (supports negative and fractional values)
     * @return base ^ exponent
     */
    public double power(double base, double exponent) {
        return Math.pow(base, exponent);
    }
}
